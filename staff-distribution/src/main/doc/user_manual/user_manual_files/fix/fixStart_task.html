fixStart

Description

fixStart task starts FIX initiator, acceptor or executor.

Parameters

Attribute Description Default Required
refid Reference name of the FIX Peer. None Yes

Examples

<target name="run_all">
	<mkdir dir="${test.log.dir}" />

	<echo>Creating FIX Initiator...</echo>
	<fixInitiator settingsfile="./settings/FIXInitiatorEmail.properties" id="main_initiator" />
	<fixAcceptor settingsfile="${fix_common.settings.dir}/FIXAcceptorEmail.properties" id="main_acceptor" />

	<fixStart refid="main_initiator" />
	<fixStart refid="main_acceptor" />
	<sleep seconds="2" />
</target>
